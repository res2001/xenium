111 void michael_scott_queue<T, Policies...>::push(T value) {
112  node* n = new node(std::move(value));
113 
114  backoff backoff;
115 
116  guard_ptr t;
117  for (;;) {
118  // Get the old _tail pointer.
119  // (3) - this acquire-load synchronizes-with the release-CAS (5, 7, 10)
120  t.acquire(_tail, std::memory_order_acquire);
121 
122  // Help update the _tail pointer if needed.
123  // (4) - this acquire-load synchronizes-with the release-CAS (6)
124  auto next = t->_next.load(std::memory_order_acquire);
125  if (next.get() != nullptr) {
126  marked_ptr expected(t.get());
127  // (5) - this release-CAS synchronizes-with the acquire-load (3)
128  _tail.compare_exchange_weak(expected, next, std::memory_order_release, std::memory_order_relaxed);
129  continue;
130  }
131 
132  // Attempt to link in the new element.
133  marked_ptr null{};
134  // (6) - this release-CAS synchronizes-with the acquire-load (2, 4, 9).
135  if (t->_next.compare_exchange_weak(null, n, std::memory_order_release, std::memory_order_relaxed)) {
136  break;
137  }
138 
139  backoff();
140  }
141 
142  // Swing the tail to the new element.
143  marked_ptr expected = t.get();
144  // (7) - this release-CAS synchronizes-with the acquire-load (3)
145  _tail.compare_exchange_strong(expected, n, std::memory_order_release, std::memory_order_relaxed);
146 }

148 template <class T, class... Policies>
149 bool michael_scott_queue<T, Policies...>::try_pop(T& result) {
150  backoff backoff;
151 
152  guard_ptr h;
153  for (;;) {
154  // Get the old _head and _tail elements.
155  // (8) - this acquire-load synchronizes-with the release-CAS (11)
156  h.acquire(_head, std::memory_order_acquire);
157 
158  // Get the _head element's successor.
159  // (9) - this acquire-load synchronizes-with the release-CAS (6).
160  auto next = acquire_guard(h->_next, std::memory_order_acquire);
161  if (_head.load(std::memory_order_relaxed).get() != h.get()) {
162  continue;
163  }
164 
165  // If the _head (dummy) element is the only one, return false to signal that
166  // the operation has failed (no element has been returned).
167  if (next.get() == nullptr) {
168  return false;
169  }
170 
171  marked_ptr t = _tail.load(std::memory_order_relaxed);
172 
173  // There are multiple elements. Help update _tail if needed.
174  if (h.get() == t.get()) {
175  // (10) - this release-CAS synchronizes-with the acquire-load (3)
176  _tail.compare_exchange_weak(t, next, std::memory_order_release, std::memory_order_relaxed);
177  continue;
178  }
179 
180  // Attempt to update the _head pointer so that it points to the new dummy node.
181  marked_ptr expected(h.get());
182  // (11) - this release-CAS synchronizes-with the acquire-load (1, 8)
183  if (_head.compare_exchange_weak(expected, next, std::memory_order_release, std::memory_order_relaxed)) {
184  // return the data of _head's successor; it is the new dummy node.
185  result = std::move(next->_value);
186  break;
187  }
188 
189  backoff();
190  }
191 
192  // The old dummy node has been unlinked, so reclaim it.
193  h.reclaim();
194 
195  return true;
196 }
